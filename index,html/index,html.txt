      function loadQuestions() {
          const container = document.getElementById('quiz-container');
          questions.forEach((item, index) => {
              const block = document.createElement('div');
              block.className = 'question-block';
              block.id = `qblock-${index}`;

              let optionsHTML = '';
              item.a.forEach((opt, optIndex) => {
                  optionsHTML += `<div class="option" onclick="selectOption(${index}, ${optIndex})" id="opt-${index}-${optIndex}">${opt}</div>`;
              });

              block.innerHTML = `
                  <div class="question-text">${index + 1}. ${item.q}</div>
                  <div class="options">${optionsHTML}</div>
              `;
              container.appendChild(block);
          });
      }

      let userAnswers = {};

      function selectOption(qIndex, optIndex) {
          userAnswers[qIndex] = optIndex;
          const options = document.querySelectorAll(`#qblock-${qIndex} .option`);
          options.forEach(opt => opt.classList.remove('selected'));
          document.getElementById(`opt-${qIndex}-${optIndex}`).classList.add('selected');
      }

      function calculateScore() {
          const name = document.getElementById('student-name').value;
          if(!name) {
              alert("Please enter your name first!");
              return;
          }

          let score = 0;
          questions.forEach((item, index) => {
              const block = document.getElementById(`qblock-${index}`);
              const selected = userAnswers[index];

              if(selected === item.correct) {
                  score++;
                  block.classList.add('correct');
              } else {
                  block.classList.add('wrong');
                  const correctText = item.a[item.correct];
                  block.innerHTML += `<span class="wrong-answer">Correct Answer: ${correctText}</span>`;
              }
          });

          const resultDiv = document.getElementById('result');
          resultDiv.style.display = 'block';
          resultDiv.innerHTML = `Student: ${name} <br> Final Score: ${score} / 100`;
          resultDiv.className = score >= 70 ? 'correct' : 'wrong';

          document.getElementById('submit-btn').style.display = 'none';
          window.scrollTo(0,0);
      }

      loadQuestions();
